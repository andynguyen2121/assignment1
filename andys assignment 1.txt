Main Algorithm:
 
1.  Get numPizzas                          // assume 1 to 5
2.  total = 0

    // part d variables

    bestIndex = -1
    bestPricePerInch = MAX_DOUBLE_VALUE

3.  for i = 1 to numPizzas
4.        Get radius[i]
5.        Get price[i]

	  // part d logic
	  
          if valuePerInch[i] < bestPricePerInch then
             bestPricePerInch = valuePerInch[i]
             bestIndex = i
          end if
8.  end for

    // part c

    delivery = 0

    if total >= 50.00 then
       delivery = 0
    else
       delivery = 8
    end if

    newTotal = total + delivery

	// part e

    display "Bill:"

    display "Grand Total: $" + newTotal

 
Subproblem — ComputeValue(radius, price):
1.  area = 3.141592 * radius * radius
2.  pricePerSquareInch = price / area
3.  return pricePerSquareInch
